    /**
 * @module iiw.core.directives
 */
define([
    'angular',
    'zeroClipboard'
], function(app, ZeroClipboard) {
    /**
     * 当用户点击元素时，把元素的data-clipboard-text属性值（变量值）复制到系统粘贴板，用于实现跨浏览器的复制功能。
     * 1. 主框架已加载，无需再次引入。
     * 2. 此类为指令，直接在html中使用，restrict: 'A'。
     * 3. 此指令必须带有data-clipboard-text属性，点击element后，把data-clipboard-text属性值（变量值）复制到系统粘贴板。
     * ***
     *     JS:
     *     app.controller('***Controller', ['$scope', function($scope) {
	 *         $scope.copytext = 'test';
	 *     }]);
     *
     *     HTML:
     *     <!-- 点击后，系统粘贴板内容为'test' -->
     *     <button i-copy-text data-clipboard-text="{{`copytext`}}">点我复制</button>
     *
     *     <!-- 点击后，系统粘贴板内容为'123' -->
     *     <button i-copy-text data-clipboard-text="123">点我复制</button>
     *
     * @class iCopyText
     *
     *
     * @author : yjj
     * @version : 1.0
     * @Date : 2015-10-29
     */

    /**
     * 当用户点击元素时，把元素的data-clipboard-text属性值（变量值）复制到系统粘贴板。
     *
     * @property data-clipboard-text
     * @type String
     */
    app.directive('iCopyText', [function() {
        return {
            restrict: 'A',
            compile: function($element) {
                new ZeroClipboard($element.get(0));
            }
        }
    }]);
});
