    /**
 * @module iiw.core.services
 */
define([
    'angular',
    'directives/message/messagepanel'
], function(app) {
    /**
     * 全局的消息提醒服务。
     * 1. 主框架已加载，无需再次引入。
     * 2. 此类为服务，需要依赖注入才能使用。
     * ***
     *      JS:
     *      app.controller('***Controller', ['iMessage', function(iMessage) {
     *          // TODO
     *      }]);
     *
     * @class iMessage
     *
     *
     * @author : yjj
     * @version : 1.0
     * @Date : 2015-11-02
     */
    app.factory('iMessage', ['$rootScope', '$compile', '$interpolate', '$filter', '$interval', 'iTimeNow', function($rootScope, $compile, $interpolate, $filter, $interval, iTimeNow) {
        var history = '';
        var _nolog = false;
        var $scope = $rootScope.$new();

        $scope.list = [];

        $('body').append('<i-message-panel></i-message-panel>');
        $compile($('i-message-panel'))($scope);

        $interval(checkSetTimeout, 1000);

        function add(json, callScope, scope) {
            removeTimeout();

            if($scope.list.length > 9) {
                $scope.list = $scope.list.slice($scope.list.length - 9);
            }

            json.level = json.level || '2';
            json.now = iTimeNow.now;
            json.time = json.time || $filter('date')(iTimeNow.now, 'yyyy-MM-dd HH:mm:ss');
            json.timeout = json.timeout || 5000;
            json.content = json.content || '';
            json.drag = json.drag || false;

            $scope.list.push({
                id: json.id || '',
                json: json,
                call: callScope,
                scope: scope,
                cs: addClass(json),
                isDrag: false
            });

            $.each($scope.list, function(i, o) {
                if(o.style) {
                    if(!o.isDrag) o.style.top += 190;
                } else {
                    o.style = {
                        right: 0,
                        top: 90
                    };

                    o.bgStyle = {
                        opacity: o.json.opacity || 1
                    }
                }
            });
        }

        function addClass(json) {
            var result = '';
            switch(json.level) {
            case 1:
            case '1':
                result = {
                    bgc: 'success',
                    icon: 'fa fa-check'
                };
                break;
            case 2:
            case '2':
                result = {
                    bgc: 'info',
                    icon: 'fa fa-info'
                };
                break;
            case 3:
            case '3':
                result = {
                    bgc: 'warning',
                    icon: 'fa fa-exclamation-triangle'
                };
                break;
            case 4:
            case '4':
                result = {
                    bgc: 'danger',
                    icon: 'fa fa-times'
                };
                break;
            }
            return result;
        }

        function checkSetTimeout() {
            $.each($scope.list, function(i, o) {
                if(o.json.timeout != '0' && o.json.now + o.json.timeout < iTimeNow.now) {
                    o.istimeout = true;
                }
            });
        }

        function removeTimeout() {
            var dels = _.where($scope.list, {istimeout: true});
            $scope.list = _.difference($scope.list, dels);
        }

        return {
            /**
             * 显示消息提示框。
             *
             * @method show
             *
             * @param json {JSON Object}
             *      json: {
             *          id: '',                     // 消息框id，默认可忽略，带有id参数的，可通过调用remove(id)方法去除消息显示。
             *          level: 1,                   // 1->成功;2->消息；3->警告；4->异常。默认为2
             *          title: '标题',              // *必填项
             *          time: '时间',               // 默认取当前客户端时间。
             *          timeout: '超时消失',        // 默认5秒（int类型）, '0'不自动消失（注意：'0'必须传递字符串）。
             *          content: '内容',           // *必填项
             *          drag: false,               // 是否允许拖动，true为允许拖动，默认不允许。
             *          opacity: 1,                // 背景透明度，默认为1，既不透明。
             *          fn: 'fn'                   // 点击消息框时响应的方法：callScope.fn，在json中传入方法名后还需传入callScope参数值：fn所属域对象
             *      }
             *
             * @param checkSame {Boolean} 默认false，当输入true时，自动过滤重复项，重复项指与上一次一样既过滤不显示。
             * @param callScope {Object} 方法触发对象，用于点击事件时调用。
             * @param scope {Object} 作用域，用于content内容为html时，触发与作用域相关的方法。
             *
             * @example
             *      // 注入iMessage服务。
             *      app.controller('***Controller', ['iMessage', function(iMessage) {
             *          // 显示一个全局的消息提醒。
             *          iMessage.show({
             *              title: '标题',
             *              content: '内容'
             *          });
             *      }]);
             */
            show: function(json, checkSame, callScope, scope) {
                if(!_nolog) {
                    if(!checkSame) {
                        add(json, callScope, scope);
                    } else {
                        if(history != json.content) {
                            add(json, callScope, scope);
                            history = json.content;
                        }
                    }
                }
            },


            /**
             * 根据消息框id，删除消息框
             *
             * @method remove
             *
             * @param id {String} 生成消息框时的id，如生成时没有配置id参数，则无法通过该方法删除消息框。
             *
             * @example
             *      // 注入iMessage服务。
             *      app.controller('***Controller', ['iMessage', function(iMessage) {
             *          // 显示一个全局的消息提醒。
             *          iMessage.show({
             *              id: 'test',
             *              title: '标题',
             *              content: '内容'
             *          });
             *
             *          // 通过代码删除消息提醒框。
             *          iMessage.remove('test');
             *      }]);
             */
            remove: function(id) {
                if(id) {
                    var o = _.findWhere($scope.list, {id: id});
                    if(o) {
                        $scope.list = _.difference($scope.list, [o]);
                    }
                }
            },


            /**
             * 暂时屏蔽或打开全局的消息显示
             *
             * @method setNoLog
             *
             * @param value {Boolean} 是否屏蔽消息提醒框弹出。
             */
            setNoLog: function(value) {
                _nolog = value;
            }
        };
    }]);
});
