    /**
 * @module iiw.core.directives
 */
define([
    'angular',
    'hammer'
], function(app, Hammer) {

    /**
     * 拖动指令，加上该指令后，元素即可拖动。
     * 1. 主框架已加载，无需再次引入。
     * 2. 此类为指令，直接在html中使用，restrict: 'A'。
     * 3. 此指令的元素需自行先配置css参数，position: absolute或relative，否则无法拖动。
     * ***
     *     HTML:
     *     <!-- 可以拖动的元素 -->
     *     <div i-drag style="position: absolute; width: 100px; height: 100px; background: yellow;">可以拖动</div>
     *
     * @class iDrag
     *
     *
     * @author : yjj
     * @version : 1.0
     * @Date : 2016-07-05
     */
    app.directive('iDrag', [function() {
        return {
            restrict: 'A',
            scope: {
                notDrag: '=dragNot',
                isDrag: '=dragIs'
            },
            link: function($scope, $element) {
                if(!$scope.notDrag) {
                    var el = $element.get(0),
                        elm = new Hammer(el),
                        startx,
                        starty,
                        movex,
                        movey,
                        boxw = $('body').width();

                    $scope.$on('$destroy', function() {
                        if(elm) {
                            elm.destroy();
                            elm = null;
                        }
                    });

                    elm.on('panstart', function() {
                        startx = parseFloat($element.css('left'));

                        if(!startx) {
                            startx = boxw - parseFloat($element.css('right')) - $element.width();
                        }

                        starty = parseFloat($element.css('top'));
                    });

                    elm.on('panmove', function(e) {
                        $element.css('left', 0);
                        $element.css('top', 0);

                        movex = startx + e.deltaX;
                        movey = starty + e.deltaY;

                        el.style.webkitTransform = 'translate3d(' + movex + 'px, ' + movey + 'px, 0)';
                        el.style.transform = 'translate3d(' + movex + 'px, ' + movey + 'px, 0)';
                    });

                    elm.on('panend', function() {
                        $element.css('left', movex);
                        $element.css('top', movey);
                        el.style.webkitTransform = 'none';
                        el.style.transform = 'none';

                        $element.attr('i-drag-move', true);

                        $scope.isDrag = true;
                    });
                }
            }
        }
    }]);
});
