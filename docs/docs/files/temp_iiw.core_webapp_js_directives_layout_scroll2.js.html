    /**
 * @module iiw.core.directives
 */
define([
    'angular',
    'iscroll2',
    'directives/layout/resize'
], function(app, IScroll) {
    /**
     * 触摸滚动条。
     * 1. 主框架已加载，无需再次引入。
     * 2. 此类为指令，直接在html中使用，restrict: 'A'。
     * 3. iScroll2使用5.2.0版本，解决iScroll内存泄露的问题。
     * 4. 不能与其它独立指令使用，如ui-layout-container，如遇到此情况请增加一级元素单独使用。
     * ***
     *        HTML:
     *        <div i-scroll2 style="height: 300px"> <!-- 必须指定高度，或上级指定高度 -->
     *            <div style="height: 900px">content</div> <!-- 此为内容，高度超出i-scroll指令元素的高度，即可触摸和鼠标滚轮进行滚动 -->
     *        </div>
     *
     * @class iScroll2
     *
     *
     * @author : yjj
     * @version : 1.0
     * @Date : 2016-09-19
     */

    /**
     * 当滚动条到底时，自动触发该属性绑定的方法。
     * 1. 通过该属性即可实现滚动分页查询功能。
     * 2. 滚动条控件采用了iScroll5，更多属性方法可参考api。
     *
     * @property data-load-event {Function}
     */
    app.directive('iScroll2', [function() {
        return {
            restrict: 'A',
            template: '<div i-resize="resize" ng-transclude style="position: absolute"></div>',
            transclude: true,
            scope: true,
            link: function($scope, $element, attr) {
                $element.css('overflow', 'hidden');

                var config = {mouseWheel: true, scrollbars: true, fadeScrollbars: true, disableMouse: false};
                if(attr.config && typeof(attr.config) == 'string') {
                    config = JSON.parse(attr.config);
                }
                var myScroll = new IScroll($element.get(0), config);
                $element.data('i-scroll', myScroll);

                $scope.resize = function() {
                    if(myScroll) myScroll.refresh();
                };

                $element.children(':first').children(':first').css('position', 'static');

                myScroll.on('scrollEnd', function() {
                    if(myScroll.y <= myScroll.maxScrollY) {
                        if(attr.loadEvent) {
                            $scope.$parent.$eval(attr.loadEvent);
                        }
                    }
                });

                $scope.$on('$destroy', function() {
                    if(myScroll) {
                        myScroll.destroy();
                        myScroll = null;
                    }
                });
            }
        }
    }]);
});
