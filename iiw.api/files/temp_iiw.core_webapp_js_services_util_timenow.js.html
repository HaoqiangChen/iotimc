    /**
 * @module iiw.core.services
 */
define(['angular'], function(app) {
    /**
     * #### *** 禁止直接使用原生JS的new Date()获取时间，而是统一通过该服务获取时间，避免出现客户端与服务端时间有较大差距而产生的错误。**
     * 获取服务器实时时间工具。
     * 1. 主框架已加载，无需再次引入。
     * 2. 此类为服务，需要依赖注入才能使用。
     * ***
     *      JS:
     *      app.controller('***Controller', ['iTimeNow', function(iTimeNow) {
     *          // TODO
     *      }]);
     *
     * @class iTimeNow
     *
     *
     * @author : yjj
     * @version : 1.0
     * @Date : 2015-10-23
     */
    app.factory('iTimeNow', ['$timeout', function($timeout) {
        var timeDifference = 0;
        var time = {};


        /**
         * 自动更新的实时时间
         *
         * @property now {Date}
         *
         * @example
         *      JS：
         *      // 注入iTimeNow服务。
         *      app.controller('***Controller', ['$scope', 'iTimeNow', function($scope, iTimeNow) {
         *          $scope.time = iTimeNow;
         *      }]);
         *
         *      HTML：
         *      <div>{{time.now | date: 'yyyy-MM-dd HH:mm:ss'}}</div> <!-- 2016-02-17 19:00:00 无需计时器，无法代码，时间自动在网页中更新 -->
         */
        (function tick() {
            time.now = new Date().getTime() + timeDifference;
            $timeout(tick, 1000);
        })();


        time['setSvrTime'] = function(timevalue) {
            if(timevalue) {
                timeDifference = timevalue - new Date().getTime();
            }
        };


        /**
         * 获取当前服务器的时间
         *
         * @method getTime
         *
         * @return {Long} 服务器当前时间，指日期和时间距1970年1月1日午夜（GMT时间）之间的毫秒数。
         *
         * @example
         *      JS：
         *      // 注入iTimeNow服务。
         *      app.controller('***Controller', ['iTimeNow', function(iTimeNow) {
         *          console.log(iTimeNow.getTime());
         *      }]);
         */
        time['getTime'] = function() {
            return new Date().getTime() + timeDifference;
        };

        return time;
    }]);
});
