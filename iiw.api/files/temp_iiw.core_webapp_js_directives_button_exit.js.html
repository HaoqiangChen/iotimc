    /**
 * @module iiw.core.directives
 */
define([
    'angular'
], function(app) {
    /**
     * 点击element，关闭浏览器，如果浏览器不支持关闭，则跳回到登陆界面。
     * 1. 主框架已加载，无需再次引入。
     * 2. 此类为指令，直接在html中使用，restrict: 'A'。
     * ***
     *        HTML:
     *        <button i-exit>关闭系统</button>
     *
     * @class iExit
     *
     *
     * @author : yjj
     * @version : 1.0
     * @Date : 2015-10-29
     */
    app.directive('iExit', ['$state', 'iConfig', function($state, iConfig) {
        return {
            restrict: 'A',
            scope: {
                state: '=exitState'
            },
            link: function($scope, $element) {
                if(!$scope.state) {
                    if(iConfig.get('webcontextpath')) {
                        $scope.state = iConfig.get('webcontextpath').content || 'login';
                    } else {
                        $scope.state = 'login';
                    }
                }

                $scope.state = $scope.state || 'login';

                $element.on('click', function() {
                    if(!$('.iiw-exit').size()) {
                        var html = [];

                        html.push('<div class="iiw-exit">');
                        html.push('<div class="layout-full table">');
                        html.push('<div class="table-cell">');
                        html.push('<div class="iiw-exit-panel">');
                        html.push('<p>确认退出系统？</p>');
                        html.push('<div class="iiw-exit-buttons">');
                        html.push('<button class="exitButton button button-glow button-rounded button-caution">确认</button>');
                        html.push('<button class="cancelButton button button-rounded button-primary">取消</button>');
                        html.push('</div>');
                        html.push('</div>');
                        html.push('</div>');
                        html.push('</div>');
                        html.push('</div>');

                        $('body').append(html.join(''));

                        setTimeout(function() {
                            $('.iiw-exit').click(hide);

                            $('.iiw-exit-panel').click(function(e) {
                                e.cancelable = true;
                                e.stopPropagation();
                            });

                            $('.iiw-exit').find('.exitButton').click(exit);

                            $('.iiw-exit').find('.cancelButton').click(hide);
                        }, 300);
                    }

                    show();
                });

                function show() {
                    $('.iiw-exit').show('fade');
                }

                function hide() {
                    $('.iiw-exit').hide('fade');
                }

                function exit() {
                    try {
                        var gui = requireNW('nw.gui'),
                            win = gui.Window.get();

                        win.close();
                    } catch(e) {
                        window.opener = null;
                        window.open(location, '_self', '');
                        window.close();

                        setTimeout(function() {
                            $state.go($scope.state);
                        }, 300);

                        hide();
                    }
                }
            }
        }
    }]);
});
