    /**
 * @module iiw.core.directives
 */
define(['angular', 'iscroll'], function(app) {
    /**
     * 触摸滚动条。
     * 1. 主框架已加载，无需再次引入。
     * 2. 此类为指令，直接在html中使用，restrict: 'A'。
     * 3. iScroll使用5.1.2版本，存在内存泄露问题，请使用iScroll2代替iScroll使用。
     * ***
     *        HTML:
     *        <div i-scroll style="height: 300px"> <!-- 必须指定高度，或上级指定高度 -->
     *            <div style="position: absolute;"> <!-- 必须采用绝对定位才可以使指令生效 -->
     *                <div style="height: 900px">content</div> <!-- 此为内容，高度超出i-scroll指令元素的高度，即可触摸和鼠标滚轮进行滚动 -->
     *            </div>
     *        </div>
     *
     * @class iScroll
     *
     *
     * @author : yjj
     * @version : 1.0
     * @Date : 2015-10-21
     */

    /**
     * 当滚动条到底时，自动触发该属性绑定的方法。
     * 1. 通过该属性即可实现滚动分页查询功能。
     * 2. 滚动条控件采用了iScroll5，更多属性方法可参考api。
     *
     * @property data-load-event {Function}
     */
    app.directive('iScroll', [function() {
        return {
            restrict: 'A',
            compile: function() {
                return {
                    post: function($scope, $element, attr) {
                        $element.css('overflow', 'hidden');
                        var config = {mouseWheel: true, scrollbars: true, fadeScrollbars: true};
                        if(attr.config && typeof(attr.config) == 'string') {
                            config = JSON.parse(attr.config);
                        }
                        var myScroll = new IScroll($element.get(0), config); // eslint-disable-line
                        $element.data('i-scroll', myScroll);

                        myScroll.on('scrollEnd', function() {
                            if(myScroll.y <= myScroll.maxScrollY) {
                                if(attr.loadEvent) {
                                    $scope.$eval(attr.loadEvent);
                                }
                            }
                        });
                    }
                }
            }
        }
    }]);
});
