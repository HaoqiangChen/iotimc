    /**
 * @module iiw.core.services
 */
define([
    'angular',
    'services/server/ajax'
], function(app) {
    /**
     * 方言管理服务。方言管理主要应用于不同行业的名称通过数据库进行配置，而无需写死代码中，例如监狱中的罪犯，在戒毒中就是学员。
     * 1. 主框架已加载，无需再次引入。
     * 2. 此类为服务，需要依赖注入才能使用。
     * ***
     *      JS:
     *      app.controller('***Controller', ['iGetLang', function(iGetLang) {
     *          // TODO
     *      }]);
     *
     * @class iGetLang
     *
     *
     * @author : yjj
     * @version : 1.0
     * @Date : 2015-11-09
     */
    app.factory('iGetLang', ['iAjax', function(iAjax) {
        var _lang = {};
        init();

        function init() {
            iAjax.getSync('/sys/web/dialect.do?action=getDialect').then(function(data) {
                _lang = data.result;
            });
        }

        return {
            /**
             * 根据key，返回目前数据库对应的方言，如当前数据库配置的行业为监狱，那么返回对被管理人员的称呼为罪犯；
             *
             * @method get
             *
             * @params key {String} 需要获取方言查询的key，参数具体根据数据库情况配置。
             *
             * @return {String} 返回key对应的方言。
             *
             * @example
             *      // 注入iGetLang服务。
             *      app.controller('***Controller', ['iGetLang', function(iGetLang) {
             *          console.log(iGetLang.get('CPN'));    // 罪犯
             *      }]);
             */
            get: function(key) {
                return _lang[key] || '';
            }
        }
    }]);
});
