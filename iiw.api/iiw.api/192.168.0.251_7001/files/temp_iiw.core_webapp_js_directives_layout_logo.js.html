    /**
 * @module iiw.core.directives
 */
define([
    'angular',
    'css!../../../css/logo.css',
    'css!../../../sys/web/logo.do'
], function(app) {
    /**
     * 系统logo及标题栏
     * 1. 主框架已加载，无需再次引入。
     * 2. 此类为指令，直接在html中使用，restrict: 'E'。
     * 3. 此指令通过数据库获取logo及标题栏配置信息。
     * ***
     *     HTML:
     *     <!-- 自动通过数据库获取配置信息，包括logo及系统名称 -->
     *     <i-logo></i-logo>
     *
     * @class iLogo
     *
     *
     * @author : yjj
     * @version : 1.0
     * @Date : 2016-02-25
     */
    /**
     * 是否不显示标题信息，默认显示，该值设置为true时不显示标题。
     * ***
     *     HTML:
     *     <i-logo logo-notitle="true"></i-logo>
     *
     * @property logo-notitle
     * @type Boolean
     */
    /**
     * 自定义主标题，开发者可以根据自己模块的需要自定义主标题值。
     * ***
     *     HTML:
     *     <!-- 注意，如果直接赋值，需要使用单引号 -->
     *     <i-logo logo-title="'主标题'"></i-logo>
     *
     * @property logo-title
     * @type String
     */
    /**
     * 自定义副标题，开发者可以根据自己模块的需要自定义副标题值。
     * ***
     *     JS:
     *     $scope.moduleName = '系统管理';
     *
     *     HTML:
     *     <!-- 注意，如果直接赋值，需要使用单引号 -->
     *     <i-logo logo-subtitle="{{moduleName}}"></i-logo>
     *
     * @property logo-subtitle
     * @type String
     */
    app.directive('iLogo', ['$timeout', 'iConfig', function($timeout, iConfig) {
        return {
            scope: {
                notitle: '=logoNotitle',
                deftitle: '=logoTitle',
                defsubtitle: '=logoSubtitle'
            },
            template: '<div class="iiw-logo-panel"><div class="logo"></div><div class="titles"><h1>{{title}}</h1><h2>{{subtitle}}</h2></div></div>',
            restrict: 'E',
            compile: function() {
                return {
                    post: function($scope, $element) {
                        $scope.title = $scope.deftitle || '　';
                        $scope.subtitle = $scope.defsubtitle || '　';

                        var _data = {
                            company: iConfig.get('company'),
                            title: iConfig.get('title'),
                            subtitle: iConfig.get('subtitle')
                        };

                        init(true);
                        $element.find('.iiw-logo-panel').parent().parent().on('resize', function() {
                            init();
                        });

                        function init(init) {
                            var panel = $element.find('.iiw-logo-panel'),
                                panel_h = panel.parent().parent().height();

                            panel.height(panel_h);
                            panel.css('font-size', panel_h);
                            var temp = (panel_h - (panel_h * 0.27 + 10 + panel_h * 0.21)) / 3;
                            panel.find('h1').css('margin-top', temp + (panel_h * 0.06 / 2));
                            panel.find('h1').css('margin-bottom', temp);

                            setValues(init);
                        }

                        function setValue(key) {
                            var result,
                                value = _data[key];
                            if(value) {
                                $scope[key] = value.content;
                                result = value.content;
                            }
                            return result;
                        }

                        function setValues(init) {
                            var company = setValue('company');
                            setLogo(company, init);

                            if(!$scope.deftitle) {
                                setValue('title');
                            }

                            if(!$scope.defsubtitle) {
                                setValue('subtitle');
                            }

                            if(!$scope.subtitle) {
                                // 如果不存在副标题时，主标题垂直居中显示
                                var panel = $element.find('.iiw-logo-panel');
                                panel
                                    .find('.titles')
                                    .height('100%')
                                    .css('display', 'table');

                                panel.find('.titles h1')
                                    .css('display', 'table-cell')
                                    .css('vertical-align', 'middle');
                            }
                        }

                        function setLogo(company, init) {
                            var panel = $element.find('.iiw-logo-panel'),
                                panel_h = panel.height(),
                                logo = $element.find('.logo'),
                                logo_w,
                                logo_h;

                            logo.removeClass(company);
                            logo.addClass(company);

                            if(init) {
                                logo_w = logo.width();
                                logo_h = logo.height();

                                logo.attr('mywidth', logo_w)
                                    .attr('myheight', logo_h);
                            } else {
                                logo_w = logo.attr('mywidth');
                                logo_h = logo.attr('myheight');
                            }

                            logo.height(panel_h);
                            logo.width(panel_h / logo_h * logo_w);
                        }

                        if($scope.notitle) {
                            $element.find('.titles').hide();
                        }
                    }
                }
            }
        }
    }]);
});
