    /**
 * 监听元素的大小变化。
 *
 * 资源优化考虑，可考虑根据list的数量来判断是否启用计时器，降低系统的资源占用。
 *
 * Created by YJJ on 2016-09-13.
 */
define([
    'angular'
], function(app) {
    /**
     * 监听元素的大小变化指令
     * 1. 主框架已加载，无需再次引入。
     * 2. 此类为指令，直接在html中使用，restrict: 'A'。
     * 3. jQuery.resize存在内存泄露，请使用指令代替。
     * ***
     *     JS:
     *     app.controller('***Controller', ['$scope', function($scope) {
	 *         $scope.resizeCallback = function($element, newWidth, newHeight, oldWidth, oldHeight) {
	 *              // TODO
	 *         };
	 *     }]);
     *
     *
     *     HTML:
     *     <!-- 当元素大小发生变化时，自动触发$scope.resizeCallback方法 -->
     *     <div i-resize="resizeCallback"></div>
     *
     * @class iResize
     *
     *
     * @author : yjj
     * @version : 1.0
     * @Date : 2016-09-19
     */
    app.directive('iResize', ['$interval', function($interval) {
        var _index = 0,
            _list = [];

        $interval(function() {
            var remove = [],
                $elem,
                data;
            $.each(_list, function(i, id) {
                $elem = $('.' + id);
                if($elem.size()) {
                    data = $elem.data('i-resize-data');
                    if(data && data.o && data.fn && data.o[data.fn] && (data.w != $elem.width() || data.h != $elem.height())) {
                        data.o[data.fn]($elem, $elem.width(), $elem.height(), data.w, data.h);

                        data.w = $elem.width();
                        data.h = $elem.height();
                    }
                } else {
                    remove.push(id);
                }
            });

            _list = _.difference(_list, remove);
        }, 100);

        return {
            restrict: 'A',
            scope: {
                fn: '@iResize'
            },
            link: function($scope, $element) {
                var id = 'i-resize-' + _index++;

                $element.data('i-resize-data', {
                    id: id,
                    w: $element.width(),
                    h: $element.height(),
                    o: $scope.$parent,
                    fn: $scope.fn
                });
                $element.addClass(id);
                _list.push(id);
            }
        }
    }]);
});
