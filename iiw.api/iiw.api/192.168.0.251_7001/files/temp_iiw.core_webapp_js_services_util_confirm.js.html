    /**
 * @module iiw.core.services
 */
define([
    'angular',
    'css!../../../css/iconfirm'
], function(app) {
    /**
     * 全局的确认弹出框服务。
     * 1. 主框架已加载，无需再次引入。
     * 2. 此类为服务，需要依赖注入才能使用。
     * ***
     *      JS:
     *      app.controller('***Controller', ['iConfirm', function(iConfirm) {
     *          // TODO
     *      }]);
     *
     * @class iConfirm
     *
     *
     * @author : yjj
     * @version : 1.0
     * @Date : 2016-06-30
     */
    app.factory('iConfirm', ['$compile', function($compile) {
        var _id = 0,
            _el;

        var defaults = {
            id: null,
            title: null,
            content: null,
            templateUrl: null,
            scope: null,
            buttons: [],
            close: null
        };

        $('body').append('<div class="iiw-confirm"></div>');

        _el = $('.iiw-confirm');

        function _show(options) {
            var settings = $.extend({}, defaults, options);

            if(settings.close) {
                $.each(settings.buttons, function(i, o) {
                    if(o && o.action && o.action == settings.close) {
                        throw 'close和button action不能为同一个方法，否则将会导致死循环！';
                    }
                });
            }

            settings.id = _id++;

            _el.append(createHtml(settings));

            var $win = $(window),
                w = $win.width(),
                h = $win.height(),
                $confirm = _el.find('#iiw-confirm-' + settings.id),
                confirm = $confirm.get(0),
                $confirmBox = $confirm.find('.iiw-confirm-box'),
                $buttons = $confirm.find('.iiw-confirm-box .iiw-confirm-box-buttons button');

            if(settings.templateUrl) {
                var $content = $confirm.find('.iiw-confirm-box .iiw-confirm-box-content');
                $content.html(getTemplate(settings.templateUrl));
            }

            if(confirm && settings.close) {
                confirm.close = function() {
                    return settings.scope.$eval(settings.close + '(' + settings.id + ')');
                };
            }

            $confirm.click(function() {
                remove($(this));
            });

            $confirmBox.click(function(e) {
                e.stopPropagation();
            });

            if(settings.scope) {
                $.each(settings.buttons, function(i, o) {
                    var button = $buttons.get(i);
                    if(button) {
                        $(button).click(function() {
                            settings.scope.$eval(o.action + '(' + settings.id + ')');
                        });
                    }
                });

                var newScope = settings.scope.$new();
                $(confirm).data('newScope', newScope);
                $compile(confirm)(newScope);
            }

            $confirm.width(w).height(h).show('fade');

            return settings.id;
        }

        function _close(id) {
            var $confirm = _el.find('#iiw-confirm-' + id);
            if($confirm.size()) {
                remove($confirm);
            }
        }

        function _getEl(id) {
            return _el.find('#iiw-confirm-' + id);
        }

        function getTemplate(url) {
            var result = '';

            $.ajax({
                url: url,
                async: false,
                cache: false,
                dataType: 'text'
            }).success(function(data) {
                result = data;
            });

            return result;
        }

        function createHtml(settings) {
            var html = [
                '<div id="iiw-confirm-' + settings.id + '" class="iiw-confirm-dialog">',
                '<div class="iiw-confirm-bg">',
                '<div class="iiw-confirm-row">',
                '<div class="iiw-confirm-box">',
                '<div class="iiw-confirm-box-title">',
                '<h1>' + settings.title + '</h1>',
                '</div>',
                '<div class="iiw-confirm-box-content">',
                '<p>' + settings.content + '</p>',
                '</div>',
                '<div class="iiw-confirm-box-buttons">',
                createButton(settings),
                '</div>',
                '</div>',
                '</div>',
                '</div>',
                '</div>'
            ];

            function createButton(settings) {
                var html = [];

                $.each(settings.buttons, function(i, button) {
                    html.push('<button class="button button-glow button-rounded ' + button.style + '">');
                    html.push(button.text);
                    html.push('</button>');
                });

                return html.join('');
            }

            return html.join('');
        }

        function remove($confirm) {
            var confirm = $confirm.get(0);

            if(confirm && confirm.close) {
                if(confirm.close()) {
                    deleteConfirm();
                }
            } else {
                deleteConfirm();
            }

            function deleteConfirm() {
                $confirm.hide('fade', function() {
                    var scope = $confirm.data('newScope');
                    if(scope) scope.$destroy();

                    $confirm.remove();
                });
            }
        }

        return {
            /**
             * 显示确认弹出框。
             *
             * @method show
             *
             * @param json {JSON Object}
             *      json: {
             *          title: '标题',              // *必填项，标题，可以是text，也可以是html
             *          content: '内容',            // *必填项，内容，可以是text，也可以是html
             *          templateUrl: '绝对地址',     // 引用的html模板地址，建议使用绝对地址，避免分布式后无法访问，此参数配置后，content无效；如：$.soa.getWebPath('iiw.safe.XXX') + '/view/test.html'
             *          scope: $scope,              // *作用域，主要用于按钮组的事件响应和关闭事件监听
             *          buttons: [],                // *按钮组，详细见下面buttons结构描述
             *          close: 'fn'                 // 关闭监听方法，scope.fn，当该方法返回false时，无法关闭确认弹出框，返回true时才能关闭
             *      }
             *
             *      buttons: [{
             *          text: '按钮内容',            // *必填项
             *          style: '按钮样式',           // 具体参考buttons样式，默认灰色；按钮样式：button-primary（蓝色）；button-action（绿色）；button-highlight（黄色）；button-caution（红色）
             *          action: 'fn'                // 按钮点击事件：scope.fn，具体使用方法看例子
             *      }]
             *
             * @return {int} 返回弹出窗口的句柄，用于关闭窗口或获取弹出框时使用
             *
             *
             * @example
             *      // 注入iMessage服务。
             *      app.controller('***Controller', ['$scope', 'iConfirm', function($scope, iConfirm) {
             *          // 显示一个确认删除的弹出确认框。
             *          iConfirm.show({
             *              scope: $scope,
             *              title: '确认删除？',
             *              content: '删除信息后将无法还原，是否确认删除？',
             *              buttons: [{
             *                  text: '确认',
             *                  style: 'button-primary',
             *                  action: 'confirmSuccess'
             *              }, {
             *                  text: '取消',
             *                  style: 'button-caution',
             *                  action: 'confirmCancel'
             *              }],
             *              close: 'confirmClose'
             *          });
             *
             *          $scope.confirmSuccess = function(id) {
             *              console.log('点击确认');
             *              iConfirm.close(id);
             *          };
             *
             *          $scope.confirmCancel = function(id) {
             *              console.log('点击取消');
             *              iConfirm.close(id);
             *          };
             *
             *          $scope.confirmClose = function(id) {
             *              console.log('弹出窗口将被关闭，如返回true，则关闭，如返回false，则无法关闭');
             *              return true;
             *          };
             *      }]);
             */
            show: _show,

            /**
             * 通过id关闭确认弹出框。
             *
             * @method close
             *
             * @param id {int} show方法返回的id，用于通过代码关闭弹出框
             *
             */
            close: _close,

            /**
             * 通过id获取弹出框的jquery对象，用于二次开发时定制使用。
             *
             * @method getEl
             *
             * @param id {int} show方法返回的id，用于通过id获取弹出框
             *
             */
            getEl: _getEl
        }
    }]);
});
